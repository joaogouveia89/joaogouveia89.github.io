import kotlin.math.abs

fun catAndMouse(x: Int, y: Int, z: Int): String {
    val distA = abs(x - z)
    val distB = abs(y - z)
    return if(distA == distB) "Mouse C"
    else if(distA < distB) "Cat A"
    else "Cat B"
}

fun main(args: Array) {

    val result = catAndMouse(1, 2, 3)

    println(result)

}
