import kotlin.collections.*

/*
* Since the question only asks the number of bribes, there's no need to really do a sorting, nor element swapping, nor "bubbling up" an element. All you need to do is to count the number of people who overtake a person.
    */

fun minimumBribes(q: Array): Unit {
    var bribes = 0
    var tooChaotic = false

    q.forEachIndexed{ idx, element ->
        if(element - (idx + 1) > 2){
            tooChaotic = true
            return@forEachIndexed
        }
        var j = if(element < 2) 0 else (element -2)
        while(j < idx){
            if(q[j] > element){
                bribes++
            } 
            j++
        }
    }

    println(if(tooChaotic) "Too chaotic" else bribes.toString())
}

fun main(args: Array) {
    val q = args.map{
        it.toInt()
    }.toTypedArray()

    minimumBribes(q)
}
