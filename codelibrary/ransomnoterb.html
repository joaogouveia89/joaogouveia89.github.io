def checkMagazine(magazine, note)
    possible = true
    map = {}
    note.each do |word|
        unless map.has_key? word
            map[word] = 1
        else
            map[word] = map[word] + 1
        end
    end

    idx = 0
    while possible && idx < magazine.count
        word = magazine[idx]
        if map.has_key? word
            map[word] = map[word] -1
        end
        idx = idx + 1
    end

    possible = map.find{ |key, value| value > 0}.nil? ? "Yes" : "No"

    puts(possible)
end
