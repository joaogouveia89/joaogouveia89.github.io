fun Array.max(): Int{
    var currentMax = get(0)
    forEach{
        if(currentMax < it){
            currentMax = it
        }
    }
    return currentMax
}

fun hurdleRace(k: Int, height: Array): Int {
    val maxHeight = height.max()
    return if((maxHeight - k) <= 0) 0 else  (maxHeight - k);
}

fun main(args: Array) {
    val height = args.map{el ->
        el.toInt() 
    }.slice(1..(args.size - 1)).toTypedArray()

    val k = args[0].toInt()

    val result = hurdleRace(k, height)

    println(result)
}
