// The already search list is important to avoid multiple searchs on repeated characters and then avoid timeouts
fun twoStrings(s1: String, s2: String): String {
    var share = false
    var idx = 0
    var s1Length = s1.length
    var alreadySearchedList = mutableListOf()
    var alreadySearched = true
    
    while(idx < s1Length && !share){
        alreadySearched = true
        
        if(s1[idx] !in alreadySearchedList){
            alreadySearchedList.add(s1[idx])
            alreadySearched = false
        }
        if(!alreadySearched && s2.indexOf(s1[idx]) != -1){
            share = true
        }
        idx++
    }

    return if(share) "YES" else "NO"
}

fun main(args: Array) {
    
    val result = twoStrings(args[0], args[1])

    println(result)
}
